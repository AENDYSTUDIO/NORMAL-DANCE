    def check_disk_space(self):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–∏—Å–∫–æ–≤–æ–≥–æ –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–∞"""

            if not os.path.exists(install_dir):
                os.makedirs(install_dir, exist_ok=True)

            total, used, free = shutil.disk_usage(install_dir)
            free_gb = free / (1024**3)

            if free_gb > 2:  # –¢—Ä–µ–±—É–µ—Ç—Å—è –º–∏–Ω–∏–º—É–º 2GB
                self.space_status.config(text=f"‚úÖ –°–≤–æ–±–æ–¥–Ω–æ: {free_gb:.1f} GB")

                self.check_results['space'] = True
                self.log_message(f"‚úÖ –°–≤–æ–±–æ–¥–Ω–æ –º–µ—Å—Ç–∞: {free_gb:.1f} GB")
            else:
                self.space_status.config(text=f"‚ùå –ú–∞–ª–æ –º–µ—Å—Ç–∞: {free_gb:.1f} GB")
                self.space_indicator.config(text="‚ùå")
                self.check_results['space'] = False
                self.log_message(f"‚ùå –ú–∞–ª–æ –º–µ—Å—Ç–∞: {free_gb:.1f} GB (—Ç—Ä–µ–±—É–µ—Ç—Å—è –º–∏–Ω–∏–º—É–º 2GB)")
        except Exception as e:
            self.space_status.config(text="‚ùå –û—à–∏–±–∫–∞ –ø—Ä–æ–≤–µ—Ä–∫–∏ –¥–∏—Å–∫–∞")
